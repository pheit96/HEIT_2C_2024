217static void control_acelerometro(void *pvParameter){
218 float valorX = 0;
219 float valorY = 0;
220 float valorZ = 0;
221 float G = 0;
222 while (true)
223 {
224 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
225 AnalogInputReadSingle(CH0, &valorX);
226 AnalogInputReadSingle(CH1, &valorY);
227 AnalogInputReadSingle(CH2, &valorZ);
228 G+=(valorX-1650)/300;
229 G+=(valorY-1650)/300;
230 G+=(valorZ-1650)/300;
231
232 if(G>4){
233 UartSendByte(UART_CONNECTOR, "Caida detectada.\r\n");
234 }
235 G=0;
236
237 }
